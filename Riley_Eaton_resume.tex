%------------- Custom Commands -------------
% create a resume ordered list item
\newcommand{\resumeItem}[1]{
  \item\small{
    {#1 \vspace{-2pt}}
  }
}

% create a resume ordered list item for projects
\newcommand{\resumeProjectItem}[1]{
  \item\footnotesize{
    {#1 \vspace{-1.5pt}}
  }
}

% create subheading text on the resume
\newcommand{\resumeSubheading}[4]{
  \vspace{-0.9pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{\small#1} & \small#2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-7pt}
}

% create a sub-subheading text on the resume
\newcommand{\resumeSubSubheading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}\vspace{-7pt}
}

% create a subheading for a project describing what was used to build it
\newcommand{\resumeProjectHeading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \small#1 & #2 \\
    \end{tabular*}\vspace{-7pt}
}
